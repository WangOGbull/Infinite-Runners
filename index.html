// index.js — Firebase Cloud Functions const functions = require('firebase-functions'); const admin = require('firebase-admin'); const { settleMatch } = require('./src/solanaClient'); admin.initializeApp(); const db = admin.database(); const FORFEIT_GRACE_SECONDS = 30; /** * Callable from the client after a match ends. Each player calls this once * with their own claimed result. Payout only fires once BOTH players' claims * exist and agree - a single client can never trigger a payout alone. */ exports.submitMatchResult = functions.https.onCall(async (data, context) => { const { roomCode, winner, matchLogHash } = data; if (!roomCode || !winner || !matchLogHash) { throw new functions.https.HttpsError('invalid-argument', 'roomCode, winner, and matchLogHash are required.'); } if (!context.auth) { throw new functions.https.HttpsError('unauthenticated', 'Must be signed in (see Firebase Auth note below).'); } const claimRef = db.ref(`rooms/${roomCode}/resultClaims/${context.auth.uid}`); await claimRef.set({ winner, matchLogHash, submittedAt: admin.database.ServerValue.TIMESTAMP }); return reconcileClaims(roomCode); }); async function reconcileClaims(roomCode) { const roomSnap = await db.ref(`rooms/${roomCode}`).once('value'); const room = roomSnap.val(); if (!room) throw new functions.https.HttpsError('not-found', 'Room not found.'); if (room.settlement?.status === 'settled') return { status: 'already_settled' }; const claims = room.resultClaims || {}; const claimValues = Object.values(claims); if (claimValues.length < 2) { return { status: 'waiting_for_opponent_claim' }; } const [a, b] = claimValues; const agree = a.winner === b.winner && a.matchLogHash === b.matchLogHash; if (!agree) { await db.ref(`rooms/${roomCode}/settlement`).set({ status: 'disputed', reason: 'Client result claims disagree - needs multisig resolve_dispute.', flaggedAt: admin.database.ServerValue.TIMESTAMP, }); return { status: 'disputed' }; } const stakeRoomId = room.stakeRoomId || parseInt(roomCode, 10); const hostPubkey = room.hostPubkey; // stashed at deposit time - see client note below const winnerPubkey = a.winner === 'host' ? hostPubkey : room.opponentPubkey; try { const signature = await settleMatch({ roomId: stakeRoomId, winnerPubkeyStr: winnerPubkey, hostPubkeyStr: hostPubkey, }); await db.ref(`rooms/${roomCode}/settlement`).set({ status: 'settled', winner: a.winner, signature, settledAt: admin.database.ServerValue.TIMESTAMP, }); return { status: 'settled', signature }; } catch (err) { console.error(`[settleMatch] room ${roomCode} failed:`, err); await db.ref(`rooms/${roomCode}/settlement`).set({ status: 'settlement_failed', error: err.message, failedAt: admin.database.ServerValue.TIMESTAMP, }); throw new functions.https.HttpsError('internal', 'On-chain settlement failed - flagged for manual review.'); } } /** * Presence-based forfeit: the client should write to * `rooms/{code}/presence/{uid}` with an `.onDisconnect()` handler that sets * `disconnectedAt`. This function watches for that write, waits out the grace * period, and if the player hasn't come back, submits a forfeit claim on * their behalf declaring the other player the winner. This is NOT the same * as an automatic refund - a disconnect during an active match is a forfeit * (opponent wins), matching your "reconnect/forfeit, not refund" requirement. */ exports.handleDisconnect = functions.database .ref('/rooms/{roomCode}/presence/{uid}/disconnectedAt') .onCreate(async (snapshot, context) => { const { roomCode, uid } = context.params; await new Promise((resolve) => setTimeout(resolve, FORFEIT_GRACE_SECONDS * 1000)); const presenceSnap = await db.ref(`rooms/${roomCode}/presence/${uid}`).once('value'); const presence = presenceSnap.val(); if (!presence || presence.reconnectedAt) return null; // they came back in time const roomSnap = await db.ref(`rooms/${roomCode}`).once('value'); const room = roomSnap.val(); if (!room || room.settlement?.status === 'settled') return null; const disconnectedIsHost = uid === room.hostUid; const forfeitWinner = disconnectedIsHost ? 'opponent' : 'host'; // A forfeit is written as if BOTH players agreed, since the disconnect // itself (plus the expired grace period) is the evidence - there's no one // left to co-sign a claim from the disconnected side. await db.ref(`rooms/${roomCode}/resultClaims`).set({ forfeit_system: { winner: forfeitWinner, matchLogHash: 'forfeit', submittedAt: admin.database.ServerValue.TIMESTAMP }, forfeit_confirm: { winner: forfeitWinner, matchLogHash: 'forfeit', submittedAt: admin.database.ServerValue.TIMESTAMP }, }); return reconcileClaims(roomCode); }); /** * Scheduled sweeper (every minute) - calls the permissionless on-chain * timeout instructions for any room past its deadline. This is a convenience, * not a dependency: players can always trigger these themselves too, since * claim_deposit_timeout / claim_settle_timeout require no special permission. * Implement the actual on-chain calls the same way stakingManager.js does - * this function only needs to iterate rooms and skip ones that aren't stale. */ exports.sweepTimeouts = functions.pubsub.schedule('every 1 minutes').onRun(async () => { const roomsSnap = await db.ref('rooms').once('value'); const rooms = roomsSnap.val() || {}; const now = Date.now() / 1000; for (const [roomCode, room] of Object.entries(rooms)) { if (!room.staking || room.settlement?.status === 'settled') continue; // TODO: read the on-chain Room account's deposit_deadline / settle_deadline // (via solanaClient.getConnection() + roomPda) and call the matching // claim_deposit_timeout / claim_settle_timeout instruction here, using // the same instruction-building approach as settleMatch() above. // Left as a direct extension point since it's mechanical, not risky - // happy to fill this in once the rest of the pipeline is confirmed working. } return null; });
